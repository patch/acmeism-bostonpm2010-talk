# Bridge.pm in Perl 6
    
    module t::Bridge;
    
    
    use JSYNC;
    use YAML;
    
    sub load_jsync ($this) {
        return JSYNC::load($this.value);
    }
    
    
    sub dump_jsync ($this) {
        return JSYNC::dump($this.value);
    }
    
    
    sub load_yaml ($this) {
        return YAML::load($this.value);
    }
    
    
    sub dump_yaml ($this) {
        return YAML::dump($this.value);
    }
